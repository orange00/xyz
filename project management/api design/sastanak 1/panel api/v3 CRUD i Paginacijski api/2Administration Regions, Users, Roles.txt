--REGIJE

--Regija ima svoj sadržaj(Content) i kategorije(Category) za svaki entitet(songs|multimedia|spirituality|proverbs|sermons|current|vocation|answers|prayers|calendar)
--Regija ima stog upravljackih korisnika s razlicitim pravima pristupa nad regijom i sadržajem regije


Region
{
	"id" : 1,
	"title":"Hrvatska salezijanska provincija",
	"description" : "Salezijanci SDB Hrvatska"
}

--dohvat svih regija
GET
/regions
RESPONSE
[Region,...]

--dohvat regije (osnovnih informacija)
GET
/regions/{id}
RESPONSE
{
	Region
}

--stvaranje regije -> automatski se stvara stog upravljackih korisnika regije
POST
/regions
REQUEST
{
	Region
}
--id=0 | undefined | null

--uredivanje regije
PUT
/regions/{id}
REQUEST
{
	Region
}

--brisanje regije -- efektivno se brišu i upravljacki korisnici regije
DELETE
/regions/{id}

--USERS
--User je upravljacki korisnik regije
--Uloge i prava pristupa:
			--FULLADMIN 
				--programer, potpuni pristup sustavu, upravljanje konfiguracijom mikroservisa, direktni pristup evictu keša,...;direktni pristup bazi,...
			--SYSTEMADMIN
				--Šibo; stvara administratore regija i System WRITERe; 
				--ovlasti nad svim sadržajem svih regija
			--SYSTEMMODERATOR
				--službeni moderatorski tim NoveEve; ovlasti nad svim sadržajem svih regija
			--REGIONADMIN
				--stvara Region moderatore; ovlasti nad tagovima pjesama pojedine regije
			--REGIONMODERATOR
				--objavljivaci sadržaja pojedine regije; ovlasti nad svim sadržajem te regije
			--DELETED
				--bivši korisnik; radi identifikacije autora sadržaja
	
USER

/users

		--REGIONADMIN -> svi korisnici regije
